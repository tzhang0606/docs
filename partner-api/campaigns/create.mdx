---
title: 'Create Campaign'
api: 'POST /v1/campaigns'
description: 'Launch a new SMS campaign'
---

Create and launch an SMS campaign to send messages to your contact list.

## Authorizations

<ParamField header="X-API-Key" type="string" required>
  Authentication header containing your API key.
</ParamField>

## Body

<ParamField body="campaign_id" type="string" required>
  Unique campaign identifier. Format: `{timestamp}x{random_18_digits}`
  
  Example: `"1763072374x123456789012345678"`
</ParamField>

<ParamField body="campaign_name" type="string" required>
  Campaign display name.
  
  Example: `"November 2024 Outreach"`
</ParamField>

<ParamField body="organization_id" type="string" required>
  Organization unique_id.
</ParamField>

<ParamField body="user_id" type="string" required>
  User unique_id (campaign creator).
</ParamField>

<ParamField body="sms_agent_id" type="string" required>
  SMS agent unique_id for conversations.
</ParamField>

<ParamField body="teli_house_number" type="string" required>
  Phone number to send from (E.164 format). Use organization's `teli_sms_number`.
  
  Example: `"+13135559999"`
</ParamField>

<ParamField body="contacts" type="array" required>
  Array of contact objects.
  
  ```json
  [
    {
      "phone_number": "+15551234567",
      "first_name": "John",
      "last_name": "Doe"
    }
  ]
  ```
</ParamField>

<ParamField body="dry_run" type="boolean">
  If true, simulates campaign without sending real SMS.
  
  Default: `false`
</ParamField>

## Response

<ResponseField name="campaign_id" type="string">
  Campaign identifier
</ResponseField>

<ResponseField name="status" type="string">
  Processing status: `processing`, `active`, `completed`
</ResponseField>

<ResponseField name="details.total_contacts" type="number">
  Number of contacts in campaign
</ResponseField>
